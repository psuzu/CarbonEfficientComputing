data <- read.csv("../data/hourly_marginal_emissions.csv")

data <- data %>%
  mutate(
    datetime = dmy_hm(datetime),
    date = as.Date(datetime),
    hour = hour(datetime)
  )

full_data <- data %>%
  complete(date, hour = 0:23)

missing_summary <- full_data %>%
  group_by(date) %>%
  summarise(missing_hours = sum(is.na(marginal_co2_rate))) %>%

clean_data <- full_data %>%
  filter(!is.na(marginal_co2_rate))

hourly_stats <- clean_data %>%
  group_by(hour) %>%
  summarise(
    mean_co2 = mean(marginal_co2_rate, na.rm = TRUE),
    median_co2 = median(marginal_co2_rate, na.rm = TRUE),
    sd_co2 = sd(marginal_co2_rate, na.rm = TRUE),
    count = n()
  ) %>%
  arrange(hour)

mean_median_plot <- ggplot(hourly_stats, aes(x = hour)) +
  geom_line(aes(y = mean_co2, color = "Mean")) +
  geom_point(aes(y = mean_co2, color = "Mean")) +
  geom_line(aes(y = median_co2, color = "Median")) +
  geom_point(aes(y = median_co2, color = "Median")) +
  labs(
    title = "Mean vs Median Carbon Intensity Over a Day",
    x = "Hour of Day",
    y = "CO2 Rate",
    color = "Metric"
  ) +
  theme_minimal()

ggplot(hourly_stats, aes(x = hour, y = mean_co2)) +
  geom_line() +
  geom_point() +
  geom_errorbar(
    aes(ymin = mean_co2 - sd_co2, ymax = mean_co2 + sd_co2),
    width = 0.2
  ) +
  labs(
    title = "Carbon Intensity with Variability",
